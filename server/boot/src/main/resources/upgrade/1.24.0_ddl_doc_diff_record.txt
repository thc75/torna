CREATE TABLE `doc_diff_record` (
	`id` BIGINT(20) NOT NULL AUTO_INCREMENT,
	`data_id` VARCHAR(64) NOT NULL COMMENT 'doc_info.data_id' COLLATE 'utf8mb4_general_ci',
	`md5_old` VARCHAR(64) NOT NULL DEFAULT '' COMMENT '旧MD5' COLLATE 'utf8mb4_general_ci',
	`md5_new` VARCHAR(64) NOT NULL DEFAULT '' COMMENT '新MD5' COLLATE 'utf8mb4_general_ci',
	`modify_source` TINYINT(4) NOT NULL DEFAULT '0' COMMENT '修改方式，0：推送，1：表单编辑',
	`modify_userId` BIGINT(20) NOT NULL DEFAULT '0' COMMENT '修改人id',
	`modify_nickname` VARCHAR(64) NOT NULL DEFAULT '' COMMENT '修改人' COLLATE 'utf8mb4_general_ci',
	`modify_type` TINYINT(255) NOT NULL DEFAULT '0' COMMENT '变更类型，0：修改，1：新增，2：删除',
	`gmt_create` DATETIME NULL DEFAULT CURRENT_TIMESTAMP,
	`gmt_modified` DATETIME NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`) USING BTREE,
	INDEX `idx_dataid` (`data_id`) USING BTREE
) COMMENT='文档比较记录' COLLATE='utf8mb4_general_ci' ENGINE=InnoDB;
